3) Functions, Decorators, Context Managers

def func(a, b=0, *args, c=1, **kw):
    """Describe params & returns"""

from functools import wraps

def retry(times=3):
    def deco(fn):
        @wraps(fn)
        def inner(*a, **k):
            for i in range(1, times+1):
                try: return fn(*a, **k)
                except Exception:
                    if i==times: raise
        return inner
    return deco

from contextlib import contextmanager
@contextmanager
def db_session(Session):
    s = Session();
    try:
        yield s; s.commit()
    except:
        s.rollback(); raise
    finally:
        s.close()

# retry with backoff (simple)
import time

def call(fn, retries=3):
    for i in range(retries):
        try: return fn()
        except Exception:
            time.sleep(2**i)
    raise

# chunk an iterable
from itertools import islice

def chunked(it, size):
    it = iter(it)
    while (batch := list(islice(it, size))):
        yield batch
