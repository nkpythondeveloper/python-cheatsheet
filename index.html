6) Functions

Args, kwargs, decorators, functools.

Args / kwargs

def greet(name: str, *, loud: bool = False) -> str:
    msg = f"Hello {name}"
    return msg.upper() if loud else msg

def demo(*args, **kwargs):
    return args, kwargs

Decorators

from functools import wraps

def timing(fn):
    @wraps(fn)
    def wrapper(*a, **k):
        import time; t=time.perf_counter();
        try:
            return fn(*a, **k)
        finally:
            print(f"{fn.__name__}: {(time.perf_counter()-t)*1e3:.2f} ms")
    return wrapper

Iterators & generators

def fib():
    a,b=0,1
    while True:
        yield a
        a,b=b,a+b

8) Exceptions & Context Managers

try/except/else/finally, with, contextlib.

Pattern

try:
    risky()
except ValueError as e:
    print("bad value", e)
else:
    print("no error")
finally:
    cleanup()

Context manager

from contextlib import contextmanager

@contextmanager
def timer():
    import time; t=time.perf_counter()
    try:
        yield
    finally:
        print(time.perf_counter()-t)

with timer():
    work()

13) Concurrency

threading, multiprocessing, asyncio.

ThreadPoolExecutor

from concurrent.futures import ThreadPoolExecutor
with ThreadPoolExecutor(max_workers=8) as ex:
    list(ex.map(lambda x: x*x, range(10)))

asyncio basics

import asyncio as aio

async def fetch(i: int):
    await aio.sleep(0.1); return i

async def main():
    results = await aio.gather(*(fetch(i) for i in range(5)))
    print(results)

aio.run(main())

sqlite3

import sqlite3 as sq
con = sq.connect(":memory:")
cur = con.cursor()
cur.execute("CREATE TABLE t(id INTEGER, name TEXT)")
cur.executemany("INSERT INTO t VALUES(?,?)", [(1,"A"),(2,"B")])
con.commit()
for row in cur.execute("SELECT * FROM t"):
    print(row)
